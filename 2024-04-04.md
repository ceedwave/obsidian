##### conda 가상환경 패키지 설치
```
# 로컬에서 conda 설치
brew install conda

# 가상환경 생성 
conda create --name 가상환경이름 python= 파이썬버전

# conda 로그인
conda activate 가상환경 이름

# conda 환경 위에 패키지 올리기
conda install 패키지명

# conda 환경 위에 python 관련 패키지 설치 시 
pip install 패키지명

# 웹 컨트롤
pip install webdriver_manager
```
